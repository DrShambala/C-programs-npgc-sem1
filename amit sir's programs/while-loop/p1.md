### Count digits
```
#include <stdio.h>
int main() {
    int n,d,c;
   printf("Enter number ");
   scanf("%d",&n);
   
   while(n>0){
       d=n%10;
       c++;
       n=n/10;
   }
   printf("number of digits are: %d",c);
    return 0;
}
```
### Even number
```
#include <stdio.h>
int main() {
    int n,a;
   printf("Enter number to print even numbers ");
   scanf("%d",&n);
   a=1;
   while(a<n){
       if(a%2==0){
           printf("%d ",a);
       }
       a++;
   }
    return 0;
}
```
