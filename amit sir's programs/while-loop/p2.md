### Sum of first and last digit of input number
```
#include <stdio.h>
int main() {
    int n,a,d,sum;
   printf("Enter number to print sum of first and last digit ");
   scanf("%d",&n);
   
   a=n%10;
   while(n>0){
       d=n%10;
       n=n/10;
   }
   sum=a+d;
   printf("%d",sum);
    return 0;
}
```
