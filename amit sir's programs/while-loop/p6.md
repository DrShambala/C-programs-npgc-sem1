## Special number
```
// Online C compiler to run C program online
#include <stdio.h>

int main() {
   int n,s=0,fact,i,d,m;
   printf("Enter number ");
   scanf("%d",&n);
   m=n;
   while(n>0){
       d=n%10;
       fact=1;
       for(i=1;i<=d;i++){
           fact=fact*i;
       }
       s=s+fact;
       n=n/10;
   }
   if(s==m){
       printf("Special number");
   }
    else{
        printf("Not Special number");
    }
    return 0;
}
```
