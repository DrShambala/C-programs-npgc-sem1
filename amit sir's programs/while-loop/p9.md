## Kaprikar Number
```
// Online C compiler to run C program online
#include <stdio.h>
#include <math.h>
int main() {
   int n,m,sq,c,x,a,b;
   printf("Enter number ");
   scanf("%d",&n);
   m=n;
   sq=n*n;
   c=0;
   while(n>0){
       c++;
       n=n/10;
   }
   x=pow(10,c);
   
   a=sq%x;
   b=sq/x;
   if(a+b==m){
       printf("Kaprikar");
   }
   else{
       printf("not Kaprikar");
   }
    return 0;
}
```
