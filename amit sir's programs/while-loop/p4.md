# Armstrong number
```
#include <stdio.h>
int main() {
    int n,c,d,i,r,s,m,x;
    printf("Enter number ");
   scanf("%d",&n);
   m=x=n;
   s=0;
    while(n>0){
       c++;
       n=n/10;
    }     // Count of digits

   while(m>0){
        d=m%10;
        r=1;
    for (i = 1; i <= c; i++) {
        r=r*d;
    }
        s=s+r;
        m=m/10;
    } 
   if(s==x){
       printf("Armstrong number");
   }
   else{
       printf("not a Armstrong number");
   }
    
    return 0;
}
```
