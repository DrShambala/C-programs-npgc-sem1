## INSERTION SORT

```
// Online C compiler to run C program online
#include <stdio.h>

int main() {
    int arr[6],i,j,t;
    // INSERTION SORT
    for(i=0;i<=5;i++){
        scanf("%d",&arr[i]);
    }
    for(i=0;i<=5;i++){
        printf("%d ",arr[i]);
    }
    printf("\n");
    for(i=0;i<=5;i++){
        for(j=0;j<=5;j++){
            if(arr[i]>=arr[j]){
                 t=arr[i];
                arr[i]=arr[j];
                arr[j]=t;
            }
           
        }
    }
    for(i=0;i<=5;i++){
        printf("%d ",arr[i]);
    }
    return 0;
}
```
