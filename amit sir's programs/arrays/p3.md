## find the maximum and minimum number of an array

```
// Online C compiler to run C program online
#include <stdio.h>

int main() {
    int arr[6],i,min,max;
    // INSERTION SORT
    for(i=0;i<=5;i++){
        scanf("%d",&arr[i]);
    }
    for(i=0;i<=5;i++){
        printf("%d ",arr[i]);
    }
    printf("\n");
    max=min=arr[0];
    
    for(i=0;i<=5;i++){
            if(max<=arr[i]){
                max=arr[i];
            }
            if(min>=arr[i]){
                min=arr[i];
            }
    }
        printf("Maximum number is %d \n",max);
        printf("Minimum number is %d \n",min);
    return 0;
}
```
